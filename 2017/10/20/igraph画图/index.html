pip的安装

下载地址

https://pypi.python.org/pypi/pip

安装

将文件放至你将要装的地方，然后打开cmd进入该文件夹下输入：
python setup.py install

python-igraph安装

首先尝试命令安装：pip install python-igraph
如果不成功则用以下办法。

下载地址

http://www.lfd.uci.edu/~gohlke/pythonlibs/#python-igraph
找到适合自己系统位数和python版本的python-igraph包下载，比如：

安装

将文件放至你将要装的地方，然后打开cmd进入该文件夹下输入：
pip install 该文件的名字
如：pip install python_igraph-0.7.1.post6-cp35-none-win_amd64.whl

脚本代码

from igraph import *
import cairo
import pymysql

conn=pymysql.connect(host='192.168.125.130',user='root',passwd='123456',db='mysql',charset='utf8')
cursor=conn.cursor()

print(conn)
print(cursor)

g = Graph()
j=0;ip=[];ip1=[];count=0;ip2=[];ips={};ss=0;ss1=0;s3=[]
try:
    # 执行SQL语句
    cursor.execute("select srcip,dstip from netflowdata")
       # 获取所有记录列表
    results = cursor.fetchall()
    for row in results:
        srcip=row[0]
        dstip=row[1]
        ip += [srcip, dstip]
        #打印结果
        print("srcip=%s,dstip=%s" % (srcip,dstip))
except:
    print("Error: unable to fetch data")
s=list(ip)
for x in s:
    if x not in ip1:
        ip1.append(x)
s1=list(ip1)
for x in s1:
    ip2+=[(x,ss)]
    ss=ss+1
    ips.update(ip2)
for i in range(len(ips)):
    count=count+1
g.add_vertices(count)
for x in s1:
    g.vs[ss1]["name"] = x
    ss1=ss1+1
g.vs["label"]=g.vs["name"]
try:
    # 执行SQL语句
    cursor.execute("select srcip,dstip from netflowdata")
    # 获取所有记录列表
    results = cursor.fetchall()
    for row in results:
        srcip=row[0]
        dstip=row[1]
        x=ips[srcip]
        y=ips[dstip]
        if (x,y) not in s3:
            s3+=[(x,y)]
except:
    print("Error: unable to fetch data")
for (x,y) in s3:
    if(x,y) and (y,x) in s3:
        s3.remove((y,x))
for (x,y) in s3:
    g.add_edges([(x,y)])
summary(g)
g.vs["size"]=30
layout=g.layout("circle")
plot(g,layout=layout)
# 关闭数据库连接
conn.close()
