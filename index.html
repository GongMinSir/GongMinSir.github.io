脚本代码

from igraph import *
import cairo
import pymysql

conn=pymysql.connect(host='192.168.125.130',user='root',passwd='123456',db='mysql',charset='utf8')
cursor=conn.cursor()

print(conn)
print(cursor)

g = Graph()
j=0;ip=[];ip1=[];count=0;ip2=[];ips={};ss=0;ss1=0;s3=[]
try:
    # 执行SQL语句
    cursor.execute("select srcip,dstip from netflowdata")
       # 获取所有记录列表
    results = cursor.fetchall()
    for row in results:
        srcip=row[0]
        dstip=row[1]
        ip += [srcip, dstip]
        #打印结果
        print("srcip=%s,dstip=%s" % (srcip,dstip))
except:
    print("Error: unable to fetch data")
s=list(ip)
for x in s:
    if x not in ip1:
        ip1.append(x)
s1=list(ip1)
for x in s1:
    ip2+=[(x,ss)]
    ss=ss+1
    ips.update(ip2)
for i in range(len(ips)):
    count=count+1
g.add_vertices(count)
for x in s1:
    g.vs[ss1]["name"] = x
    ss1=ss1+1
g.vs["label"]=g.vs["name"]
try:
    # 执行SQL语句
    cursor.execute("select srcip,dstip from netflowdata")
    # 获取所有记录列表
    results = cursor.fetchall()
    for row in results:
        srcip=row[0]
        dstip=row[1]
        x=ips[srcip]
        y=ips[dstip]
        if (x,y) not in s3:
            s3+=[(x,y)]
except:
    print("Error: unable to fetch data")
for (x,y) in s3:
    if(x,y) and (y,x) in s3:
        s3.remove((y,x))
for (x,y) in s3:
    g.add_edges([(x,y)])
summary(g)
g.vs["size"]=30
layout=g.layout("circle")
plot(g,layout=layout)
# 关闭数据库连接
conn.close()

netflow介绍

  NetFlow是一种数据交换方式，其工作原理是：NetFlow利用标准的交换模式处理数据流的第一个IP包数据，生成NetFlow 缓存，随后同样的数据基于缓存信息在同一个数据流中进行传输，不再匹配相关的访问控制等策略，NetFlow缓存同时包含了随后数据流的统计信息。
